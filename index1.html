import json import os import re from js import document, localStorage, setTimeout, window from pyodide.ffi import create_proxy # ملف تخزين بيانات الأطباء DOCTORS_FILE = "doctors.json" def load_doctors(): """تحميل بيانات الأطباء من الملف""" try: if os.path.exists(DOCTORS_FILE): with open(DOCTORS_FILE, 'r', encoding='utf-8') as f: return json.load(f) return {} except Exception as e: console.error(f"خطأ في تحميل البيانات: {str(e)}") return {} def save_doctors(doctors): """حفظ بيانات الأطباء في الملف""" try: with open(DOCTORS_FILE, 'w', encoding='utf-8') as f: json.dump(doctors, f, ensure_ascii=False, indent=4) return True except Exception as e: console.error(f"خطأ في حفظ البيانات: {str(e)}") return False def show_message(element_id, message, message_type): """عرض رسالة للمستخدم""" message_div = document.getElementById(element_id) message_div.textContent = message message_div.className = f"message {message_type}" message_div.classList.remove("hidden") setTimeout(create_proxy(lambda: message_div.classList.add("hidden")), 5000) def validate_email(email): """التحقق من صحة البريد الإلكتروني""" return re.match(r"[^@]+@[^@]+\.[^@]+", email) is not None def validate_name(name): """التحقق من صحة الاسم (أحرف فقط)""" # اسم يجب أن يحتوي على أحرف فقط (مسموح بالمسافات والأحرف العربية) return re.match(r"^[\u0600-\u06FF\s]+$", name) is not None def validate_phone(phone): """التحقق من صحة رقم الهاتف (أرقام فقط)""" return phone.isdigit() def validate_password(password): """التحقق من صحة كلمة المرور""" return len(password) >= 6 def show_error(element_id, message): """عرض رسالة خطأ تحت الحقل""" error_div = document.getElementById(element_id) error_div.textContent = message error_div.style.display = "block" def hide_error(element_id): """إخفاء رسالة الخطأ""" error_div = document.getElementById(element_id) error_div.style.display = "none" def register_doctor(): """تسجيل طبيب جديد""" # إخفاء جميع رسائل الخطأ hide_error("name-error") hide_error("phone-error") hide_error("email-error") hide_error("password-error") hide_error("confirm-error") name = document.getElementById("reg-name").value.strip() phone = document.getElementById("reg-phone").value.strip() email = document.getElementById("reg-email").value.strip() password = document.getElementById("reg-password").value.strip() confirm = document.getElementById("reg-confirm").value.strip() # التحقق من صحة البيانات valid = True if not name: show_error("name-error", "يجب إدخال اسم الطبيب") valid = False elif not validate_name(name): show_error("name-error", "الاسم يجب أن يحتوي على أحرف فقط ولا يقبل الأرقام") valid = False if not phone: show_error("phone-error", "يجب إدخال رقم الهاتف") valid = False elif not validate_phone(phone): show_error("phone-error", "رقم الهاتف يجب أن يتكون من أرقام فقط") valid = False if not email: show_error("email-error", "يجب إدخال البريد الإلكتروني") valid = False elif not validate_email(email): show_error("email-error", "بريد إلكتروني غير صالح") valid = False if not password: show_error("password-error", "يجب إدخال كلمة المرور") valid = False elif not validate_password(password): show_error("password-error", "كلمة المرور يجب أن تكون 6 أحرف على الأقل") valid = False if not confirm: show_error("confirm-error", "يجب تأكيد كلمة المرور") valid = False elif password != confirm: show_error("confirm-error", "كلمة المرور غير متطابقة") valid = False if not valid: return doctors = load_doctors() if name in doctors: show_message("register-message", "هذا الطبيب مسجل بالفعل", "warning") return # تسجيل الطبيب الجديد doctors[name] = { 'phone': phone, 'email': email, 'password': password # في التطبيق الحقيقي يجب تشفير كلمة المرور } if save_doctors(doctors): show_message("register-message", "تم التسجيل بنجاح! يمكنك الآن تسجيل الدخول", "success") # مسح الحقول بعد التسجيل document.getElementById("reg-name").value = "" document.getElementById("reg-phone").value = "" document.getElementById("reg-email").value = "" document.getElementById("reg-password").value = "" document.getElementById("reg-confirm").value = "" # الانتقال لتبويب تسجيل الدخول switch_tab("login") else: show_message("register-message", "حدث خطأ أثناء التسجيل", "error") def login_doctor(): """تسجيل دخول الطبيب""" name = document.getElementById("login-name").value.strip() password = document.getElementById("login-password").value.strip() if not name or not password: show_message("login-message", "يجب إدخال اسم المستخدم وكلمة المرور", "error") return doctors = load_doctors() if name not in doctors: show_message("login-message", "اسم المستخدم غير صحيح", "error") return if doctors[name]['password'] != password: show_message("login-message", "كلمة المرور غير صحيحة", "error") return # تسجيل الدخول الناجح show_message("login-message", f"مرحباً دكتور {name}! جاري تحويلك للنظام...", "success") # تخزين بيانات الجلسة localStorage.setItem("loggedIn", "true") localStorage.setItem("doctorName", name) localStorage.setItem("doctorEmail", doctors[name]['email']) # التوجيه إلى صفحة النظام الخبير بعد تأخير setTimeout(create_proxy(lambda: window.location.assign("index3.html")), 2000) def switch_tab(tab_id): """تبديل بين علامات التبويب""" # إخفاء جميع المحتويات tabs = document.querySelectorAll(".tab-content") for tab in tabs: tab.classList.remove("active") # إظهار المحتوى المطلوب document.getElementById(tab_id).classList.add("active") # تحديث الأزرار النشطة tab_btns = document.querySelectorAll(".tab-btn") for btn in tab_btns: btn.classList.remove("active") if btn.getAttribute("data-tab") == tab_id: btn.classList.add("active") # إخفاء جميع الرسائل document.getElementById("login-message").classList.add("hidden") document.getElementById("register-message").classList.add("hidden") # إخفاء جميع رسائل الأخطاء hide_error("name-error") hide_error("phone-error") hide_error("email-error") hide_error("password-error") hide_error("confirm-error") def setup_event_listeners(): """تهيئة مستمعي الأحداث""" # أحداث أزرار التبويب tab_btns = document.querySelectorAll(".tab-btn") for btn in tab_btns: btn.addEventListener("click", create_proxy(lambda e: switch_tab(e.target.getAttribute("data-tab")))) # أحداث الروابط المساعدة document.getElementById("switch-to-register").addEventListener("click", create_proxy(lambda e: (e.preventDefault(), switch_tab("register")))) document.getElementById("switch-to-login").addEventListener("click", create_proxy(lambda e: (e.preventDefault(), switch_tab("login")))) # أحداث الأزرار الرئيسية document.getElementById("register-btn").addEventListener("click", create_proxy(lambda e: register_doctor())) document.getElementById("login-btn").addEventListener("click", create_proxy(lambda e: login_doctor())) # إدخال البيانات بالضغط على Enter document.getElementById("login-password").addEventListener("keypress", create_proxy(lambda e: e.key == "Enter" and login_doctor())) document.getElementById("reg-confirm").addEventListener("keypress", create_proxy(lambda e: e.key == "Enter" and register_doctor())) # التحقق من صحة الاسم أثناء الكتابة document.getElementById("reg-name").addEventListener("input", create_proxy(lambda e: hide_error("name-error") if validate_name(e.target.value) else None ) ) # التحقق من صحة رقم الهاتف أثناء الكتابة document.getElementById("reg-phone").addEventListener("input", create_proxy(lambda e: hide_error("phone-error") if validate_phone(e.target.value) else None ) ) # التحقق من صحة كلمة المرور أثناء الكتابة document.getElementById("reg-password").addEventListener("input", create_proxy(lambda e: hide_error("password-error") if validate_password(e.target.value) else None ) ) # التحقق من تطابق كلمة المرور أثناء الكتابة document.getElementById("reg-confirm").addEventListener("input", create_proxy(lambda e: hide_error("confirm-error") if e.target.value == document.getElementById("reg-password").value else None ) ) # تهيئة النظام عند التحميل setup_event_listeners() console.log("تم تحميل نظام تسجيل الأطباء بنجاح")
